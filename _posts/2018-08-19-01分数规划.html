由于这是一个完全图，我写的是Floyd。判负环只需检查d[i][i]。SPFA似乎跑得快一些。

#include <cstdio>
#include <algorithm>
using namespace std;
const int MAX_N = 100;
const double pre = 0.0001, eps = 1e-9, inf = 1e20;
int n, a[MAX_N][MAX_N], b[MAX_N][MAX_N];
double d[MAX_N][MAX_N];

int sgn(double x)
{
    return x > eps ? 1 : (x < -eps ? -1 : 0);
}

double g(double l)
{
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            d[i][j] = l*b[i][j] - a[i][j];

    for (int k = 0; k < n; ++k)
        for (int i = 0; i < n; ++i)
            if (i != k)
                for (int j = 0; j < n; ++j)
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
    for (int i = 0; i < n; ++i)
        if (d[i][i] < 0)
            return inf;
    return -d[0][n-1];
}

int main()
{
    scanf("%d", &n);
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            scanf("%d", &a[i][j]);
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j)
            scanf("%d", &b[i][j]);

    double l = 0, r = 1000000;
    while (r - l > pre) {
        double m = (l+r)/2;
        int s = sgn(g(m));

        if (s < 0)
            r = m;
        else if (s > 0)
            l = m;
        else
            l = r = m;
    }

    printf("%.3f\n", (l+r)/2);

    return 0;

}
